29 class CallGraph:
30  """
31  The CallGraph class provides a directed bipartite graph of functions and
32  calls between them.
33 
34  A call graph represents the data of a @c pstats.Stats instance: it
35  interprets the functions and their mutual invocations in the profile as
36  nodes (see Function and Call). All profiling data is accessible via
37  respective node methods.
38 
39  The graph is easy to traverse both forwards (from caller to callee)
40  and backwards (from callee to caller). Its purpose is to support
41  operations that are difficult to express in the list-sorting style
42  of @c pstats.Stats. Examples of such operations are pruning of execution
43  branches with treat_as_builtin(), and merging of nodes with
44  treat_as_inlined().
45 
46  Multiple @c pstats.Stats objects can be merged into a single call graph
47  by invoking add() several times.
48 
49  @see Call, Function, and @c pstats.Stats in the
50  Python library documentation.
51  """
52  def __init__(self, stats=None):
53  """
54  Constructs a CallGraph from the given @p stats object.
55 
56  @param stats The @c pstats.Stats compatible object whose
57  data is to be represented as the new CallGraph.
58  """
59  # Function -> ( Outgoing Calls, Incoming Calls )
60  self.__functions = {}
61  # Call -> ( Calling Function, Called Function )
62  self.__calls = {}
63 
64  # Indexes to look up Functions
65  self.__fln_index = {} # (filename, line number, name) -> Function
66  self.__namespace_index = {} # namespace name -> set of Functions
67  if stats:
68  self.add( stats )

71  def add(self, stats):
72  """
73  Import data from a @c pstats.Stats compatible object.
74 
75  Invoke the method on every source that should be added to the graph.
76  The data will be merged with functions discerned by name, file name,
77  and line number: only if all three are identical will functions be
78  considered the same and their profiles combined.
79 
80  @param stats The @c pstats.Stats compatible object whose
81  data is to be added to the CallGraph.
82  """
83  # fln = (filename, line_number, name)
84  for fln, data in stats.stats.items():
85  function = self._setdefault_function( fln )
86 
87  # Dictionary of all functions that called 'function'
88  caller_dict = data[4]
89  for caller_fln, caller_data in caller_dict.items():
90  caller = self._setdefault_function( caller_fln )
91  self._add_call( caller, function, *caller_data )
92 
93  # The root function has no incoming calls to store its data.
94  if not caller_dict:
95  function.add_primitive_callcount_offset( data[0] )
96  function.add_total_callcount_offset( data[1] )
97  function.add_inline_time_offset( data[2] )
98  # Cumulative time is calculated from outgoing calls.
99

102  def total_time(self):
103  """
104  Returns the duration of the profiling.
105 
106  The duration is the amount of time that passed during the profiling.
107  Every amount of time spent in individual functions or execution
108  branches is less or equal to this duration.
109 
110  @return The amount of time that passed during the profiling.
111  """
112  cumulative_times = [ f.cumulative_time() for f in self ]
113  return reduce( max, cumulative_times, 0 )

161  def treat_as_builtin(self, function):
162  """
163  Make @p function behave like a built-in function.
164 
165  The method removes the function from the graph; the profiling data,
166  execution time and call counts, will be merged with the calling and
167  called functions.
168 
169  Use this method to prune the graph, for example when hiding irrelevant
170  functions. It preserves the execution profile.
171 
172  @param function The Function to hide.
173  """
174  for call in function.incoming_calls():
175  call.caller().add_inline_time_offset( call.cumulative_time() )
176  for call in function.outgoing_calls():
177  call.callee().add_inline_time_offset( call.inline_time() )
178  call.callee().add_primitive_callcount_offset( call.primitive_callcount() )
179  call.callee().add_total_callcount_offset( call.total_callcount() )
180  self._remove_function( function )

183  def treat_as_inlined(self, function):
184  """
185  Merge @p function with its callers, thereby pretending that it was an
186  inline function.
187 
188  This method is handy for eliminating wrappers from the graph.
189 
190  The inline_time() of @p function will be added to its callers' inline
191  execution times. Its outgoing calls will also be added to its callers.
192 
193  However, be cautious when using this method: it cannot separate
194  execution paths and therefore duplicates all outgoing calls. Suppose
195  Function C is called by A and B; C invokes D and E. After inlining C,
196  the graph will contain Calls from A to D and E, as well as from B to D
197  and E.
198 
199  @param function The Function to be inlined into its callers.
200  """
201  for in_call in function.incoming_calls():
202  in_call.caller().add_inline_time_offset( in_call.inline_time() )
203  for out_call in function.outgoing_calls():
204  self._add_call(
205  in_call.caller(),
206  out_call.callee(),
207  out_call.primitive_callcount(),
208  out_call.total_callcount(),
209  out_call.inline_time(),
210  out_call.cumulative_time()
211  )
212  self._remove_function( function )

215  def treat_as_same(self, function_a, function_b, result_name=None ):
216  """
217  Merge two Functions, pretending that both represent the same function.
218 
219  This method can help to simplify profiles with functions in many
220  different namespaces.
221 
222  The method removes both functions from the graph and creates a new
223  one that represents their combination. All incoming and outgoing
224  calls to @p function_a and @p function_b will be combined into
225  incoming and outgoing calls of the merged function. The execution
226  times and call counts will be summed up if parallels arise.
227 
228  @return The merged Function.
229  """
230  if not result_name:
231  result_name = function_a.absolute_name() + "*"
232 
233  merged_function = self._add_function( ("<merge>", 0, result_name) )
234 
235  self._merge_function( function_a, merged_function )
236  self._merge_function( function_b, merged_function )
237 
238  return merged_function

241  def merge_namespaces(self, namespaces, result_namespace):
242  """
243  Merge a list of @p namespaces into a single @p result_namespace.
244 
245  All functions with the same Function.name() in the @p namespaces will
246  unified into a function in @p result_namespace. Execution times and
247  call counts will be summarized. If the function does not yet exist,
248  it will be added to the call graph; existing functions will simply have
249  the costs added.
250 
251  @param namespaces A list of namespace name strings, or a
252  string with a regular expression.
253  @param result_namespace Name of the namespace to contain the
254  merged functions.
255  """
256  if isinstance( namespaces, str ):
257  try:
258  namespace_re = re.compile( namespaces )
259  namespaces = [ ns for ns in self.namespaces()
260  if namespace_re.match( ns ) ]
261  if namespaces:
262  result_namespace = \
263  result_namespace.format(
264  **namespace_re.match( namespaces[0] ).groupdict()
265  )
266 
267  except AttributeError:
268  message = "expected iterable or regular expression pattern"
269  raise ValueError( message )
270 
271  # Lists of all functions with the same name
272  source_index = {}
273  for ns_name in namespaces:
274  for function in self.namespace( ns_name ):
275  source_index.setdefault( function.name(), [] ).append( function )
276 
277  target_index = {}
278  for function in self.namespace( result_namespace ):
279  target_index[ function.name() ] = function
280 
281  for name, function_list in source_index.items():
282  # Not updating the target index is OK: every name appears only once
283  if name in target_index:
284  target_function = target_index[ name ]
285  else:
286  result_name = "{0}::{1}".format( result_namespace, name )
287  file_line_number = ( "<merge>", 0, result_name )
288  target_function = self._add_function( file_line_number )
289 
290  for source_function in function_list:
291  self._merge_function( source_function, target_function )

312  def _add_function(self, file_line_name):
313  """
314  Add a Function identified by @p file_line_name to the call graph. Raise
315  a @c ValueError if it already exists.
316 
317  @param file_line_name A tuple @c (filename, line_number, name)
318  uniquely identifying the Function.
319  @exception ValueError A function with this tuple already exists.
320 
321  @return The fresh Function instance created from the
322  @p file_line_name tuple.
323  """
324  if not file_line_name in self.__fln_index:
325  function = Function( self, *file_line_name )
326  # ( Outgoing Calls, Incoming Calls )
327  self.__functions[ function ] = ( set(), set() )
328 
329  # Update indexes
330  self.__fln_index[ file_line_name ] = function
331  self.__namespace_index.setdefault( function.namespace(), set() ).add( function )
332 
333  return function
334  message = "Function with this identifying triple already exists: {0}"
335  raise ValueError( message.format( file_line_name ) )

338  def _remove_function(self, function):
339  """
340  Remove @p function from the call graph together with all incoming and
341  outgoing Calls.
342 
343  @param function The Function to remove from the graph.
344  """
345  # Remove from indexes
346  fln = ( function.filename(), function.line_number(), function.absolute_name() )
347  del self.__fln_index[ fln ]
348 
349  self.__namespace_index[ function.namespace() ].discard( function )
350  if not self.__namespace_index[ function.namespace() ]:
351  del self.__namespace_index[ function.namespace() ]
352 
353  # Remove all associated Calls
354  for calls in self.__functions[ function ]:
355  # Removing calls mutates the set; copy it to avoid confusion.
356  for call in calls.copy():
357  self._remove_call( call )
358 
359  # Remove Function
360  assert( self.__functions[ function ] == ( set(), set() ) )
361  del self.__functions[ function ]
362  function._invalidate()

365  def _merge_function(self, source_function, target_function):
366  """
367  Merge @p source_function into @p target_function.
368 
369  The method removes the source function from the graph; all its incoming
370  and outgoing calls will be added to the incoming and outgoing calls of
371  the target function. The execution times and call counts will be summed
372  up if parallels arise.
373  """
374 
375  # Merge offsets
376  it_offset = source_function.inline_time_offset()
377  pc_offset = source_function.primitive_callcount_offset()
378  tc_offset = source_function.total_callcount_offset()
379  target_function.add_inline_time_offset( it_offset )
380  target_function.add_primitive_callcount_offset( pc_offset )
381  target_function.add_total_callcount_offset( tc_offset )
382 
383  # Merge incoming calls; _add_call merges parallel calls
384  for call in source_function.incoming_calls().copy():
385  self._add_call(
386  call.caller(),
387  target_function,
388  call.primitive_callcount(),
389  call.total_callcount(),
390  call.inline_time(),
391  call.cumulative_time()
392  )
393  self._remove_call( call )
394 
395  # Merge outgoing calls
396  for call in source_function.outgoing_calls().copy():
397  self._add_call(
398  target_function,
399  call.callee(),
400  call.primitive_callcount(),
401  call.total_callcount(),
402  call.inline_time(),
403  call.cumulative_time()
404  )
405  self._remove_call( call )
406 
407  assert not source_function.incoming_calls()
408  assert not source_function.outgoing_calls()
409  self._remove_function( source_function )

412  def _add_call(self, caller, callee, primitive_callcount, total_callcount, inline_time, cumulative_time ):
413  """
414  Add a Call to the graph with the given profiling data between @p caller
415  and @p callee.
416 
417  @return The freshly constructed Call instance.
418 
419  @see Call for a description of parameters.
420  """
421  existing_calls = caller.outgoing_calls().intersection(
422  callee.incoming_calls()
423  )
424  assert len( existing_calls ) <= 1, \
425  "call graph contains parallel Calls"
426 
427  if existing_calls:
428  ec = existing_calls.pop()
429  call = Call( self,
430  primitive_callcount + ec.primitive_callcount(),
431  total_callcount + ec.total_callcount(),
432  inline_time + ec.inline_time(),
433  cumulative_time + ec.cumulative_time()
434  )
435  self._remove_call( ec )
436  else:
437  call = Call( self,
438  primitive_callcount,
439  total_callcount,
440  inline_time,
441  cumulative_time
442  )
443  self.__calls[ call ] = ( caller, callee )
444  self._outgoing_calls( caller ).add( call )
445  self._incoming_calls( callee ).add( call )
446  return call

503 class CallGraphNode:
504  """
505  Base class for nodes of CallGraphs.
506 
507  The class provides the mechanisms that ensure correct ownership of nodes:
508  all nodes belong to a unique CallGraph that manages all edge information.
509  If a node wants to know its neighbors, it asks its owning CallGraph (for
510  example via @c CallGraph._caller() and the like).
511 
512  When the node is removed, the graph invokes @c _invalidate(). Afterwards,
513  _assert_valid() raises a @c ValueError.
514  New node types should inherit from this class.
515  """
516  def __init__(self, callgraph):
517  """
518  Constructor for new nodes. It registers the @p callgraph as owner of
519  the new node.
520 
521  Subclasses must invoke this constructor, for example via
522  @c super().__init__(callgraph)
523 
524  @param callgraph The CallGraph to which the new node belongs.
525  """
526  self.__callgraph = callgraph
527 
528  def valid(self):
529  """
530  @return @c True, if the node still belongs to its owning CallGraph;
531  @c False otherwise.
532  """
533  return self.__callgraph is not None
534 
535  def _assert_valid(self):
536  """
537  Ascertain that the node is still element of its owning CallGraph.
538  If that is not the case, raise a @c ValueError.
539 
540  Use this method as a guard in member functions operating on the
541  graph (such as retrieving neighbors).
542  """
543  if self.valid():
544  return
545  raise ValueError( "this element is no longer part of a CallGraph" )
546 
547  def _callgraph(self):
548  """
549  @return The CallGraph that owns this node.
550  """
551  return self.__callgraph
552 
553  def _invalidate(self):
554  """
555  Signal the node that it has been deleted from its graph.
556  """
557  self.__callgraph = None

564 class Function( CallGraphNode ):
565  """
566  Function objects represent called functions in a CallGraph; they originate
567  in @c pstats.Stats entries and thus are identified by filename, line number,
568  and function name.
569 
570  The objects track their incoming and outgoing calls. It is therefore
571  possible to traverse the graph along execution paths. Direct traversal to
572  adjacent Functions is available from the caller() and callee() convenience
573  methods.
574 
575  Lack of detail on the side of the @c cProfile module, however, makes it
576  impossible to separate specific execution paths. Some information on the
577  timing of a path is available from cumulative_time() in incoming_calls()
578  because these values are the measured time it took to execute the function
579  body and all invoked functions.
580 
581  The class further provides convenience methods to retrieve (accumulated)
582  profiling information for the function. To get information about a specific
583  invocation path, use Call objects.
584 
585  @see Call and CallGraph
586  """
587  def __init__(self, callgraph, filename, line_number, name):
588  """
589  Construct a new Function without connections.
590 
591  @param callgraph The CallGraph instance to which this node
592  belongs. (See CallGraphNode.)
593  @param filename Name of the file that contains the
594  represented function's source code.
595  @param line_number Number of the first line of function code in
596  above file.
597  @param name The function name. If @p name contains a double
598  color (@c ::), the last portion after the final
599  occurence will be interpreted as the function
600  name. The portion on the left will be the
601  function's namespace. (Thus, @c ns1::ns2::f
602  as @p name results in a name() of @c f and a
603  namespace() of @c ns1::ns2.
604  """
605  super().__init__( callgraph )
606  self.__filename = filename
607  self.__line_number = line_number
608  if "::" in name:
609  self.__namespace, self.__name = name.rsplit( "::", 1 )
610  else:
611  self.__namespace, self.__name = "", name
612  self.__inline_time_offset = 0
613  self.__primitive_callcount_offset = 0
614  self.__total_callcount_offset = 0

677  def inline_time(self):
678  """
679  Retrieve the total time spent executing the function's body, ignoring
680  the time spent in calls to other functions.
681 
682  This method sums the information over all execution paths. Use
683  incoming_calls() to get the times of specific invocations.
684 
685  Use add_inline_time_offset() to add execution time that does not appear in
686  the graph (for example when deleting or merging Calls).
687 
688  This method raises a ValueError if the Function is no longer part of
689  its owning CallGraph; see CallGraphNode.
690 
691  @exception ValueError The Function was already deleted from the call
692  graph.
693 
694  @return The execution time for expressions in the function body.
695  Time spent in called functions does not add to this time.
696 
697  @see add_inline_time_offset()
698  """
699  inline_times = [ c.inline_time() for c in self.incoming_calls() ]
700  return reduce( add, inline_times, self.__inline_time_offset )
701 
702  def cumulative_time(self):
703  """
704  Retrieve the total time spent executing the function. This includes
705  the time spent in called functions.
706 
707  Times are accumulated over all execution paths. Use incoming_calls()
708  to access information of a specific invocation.
709 
710  This method raises a ValueError if the Function is no longer part of
711  its owning CallGraph; see CallGraphNode.
712 
713  @exception ValueError The Function was already deleted from the call
714  graph.
715 
716  @return The total time it took to execute this function and all
717  the functions it called.
718  """
719  outgoing_times = [ c.cumulative_time() for c in self.outgoing_calls() ]
720  return reduce( add, outgoing_times, self.inline_time() )

722  def primitive_callcount(self):
723  """
724  Retrieve the number of times this function was called, not counting
725  invocations via recursive loops.
726 
727  Call counts are added up over all execution paths. Use incoming_calls()
728  to access information of a specific invocation. If you plan to remove
729  calls from the graph and nevertheless want to preserve the call count,
730  use add_primitive_callcount_offset() to introduce an offset.
731 
732  This method raises a ValueError if the Function is no longer part of
733  its owning CallGraph; see CallGraphNode.
734 
735  @exception ValueError The Function was already deleted from the call
