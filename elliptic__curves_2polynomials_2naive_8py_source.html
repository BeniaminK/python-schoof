32 @operand_casting

34 @profiling_name( "{_elliptic_curve}[x,y]" )
35 class CurvePolynomials( CommutativeRing, metaclass=template("_elliptic_curve") ):
36  """
37  A ring of polynomials over an elliptic curve; the polynomials support infix
38  notation for ring operations, as well as mixed argument types.
39 
40  This is a template class that must be instantiated with the elliptic curve.
41  Use it, for example, as follows:
42  @code
43  # Create the underlying elliptic curve
44  E = elliptic_curves.naive.EllipticCurve( FiniteField(7), 3, 4 )
45  # Instantiate the template; GF7E is a class (here: polynomials over E)
46  GF7E = CurvePolynomials( E )
47 
48  # Use two coefficient lists as arguments; the polynomial is (list1) + y*(list2)
49  s = GF7E( (4, 3, 0, 1) ) # The curve's defining polynomial
50  y = GF7E( 0, (0, 1) ) # Another polynomial: y
51  y**2 == s # The fundamental relation holds
52  type(s) is GF7E # This is also true
53 
54  # Use the polynomials 'x' and 'y' for a more natural syntax
55  x = GF7( (0, 1), () ) # The empty tuple is just for emphasis
56  y = GF7( (), (0, 1) ) # The empty tuple is just for emphasis
57  y**2 == x**3 + 3**x + 4
58  p = x**2 + 2* x + y*( 3 * x**5 + 6 * x**2 ) # A (pseudo) random polynomial
59  @endcode
60 
61  The ring of polynomials over an elliptic curve @f$ E @f$ is
62  @f$ \mathbb{F}[x,y]/(y^2 - x^3 - Ax - B) @f$, where @f$ \mathbb{F} @f$ is
63  the field over which the curve was defined, and @f$ A, B @f$ are the curve
64  parameters. The quotient polynomial @f$ y^2 - x^3 - Ax - B @f$ expresses
65  that the polynomial's domain is the set of points on @f$ E @f$; the
66  (Cartesian) coordinates of the points @f$ (x, y) \in E @f$ are connected by
67  the fundamental relation @f$ y^2 = x^3 + Ax + B @f$.
68 
69  @note As a consequence of the fundamental relation, powers of @f$ y @f$
70  greater than one can be reduced to a polynomial in @f$ x @f$. This
71  is what this implementation does. Since it does not provide full
72  multivariate arithmetic, division (with remainder; see __divmod__())
73  only works in the following cases:
74  - The divisor is a polynomial in @f$ x @f$ alone; this works for all
75  dividends.
76  - Dividend and divisor are polynomials in @f$ y @f$ alone.
77 
78  Besides restricting arithmetic, the implementation emphasizes
79  simplicity over speed; it omits possible optimizations.
80 
81  @note The class uses the operand_casting() decorator: @c other operands in
82  binary operations will first be treated as CurvePolynomial elements.
83  If that fails, the operation will be repeated after @p other was fed
84  to the constructor __init__(). If that fails, too, then the
85  operation returns @c NotImplemented (so that @p other.__rop__()
86  might be invoked).
87 
88  @see Charlap, Leonard S. and Robbins, David P., "CRD Expositroy Report 31:
89  An Elementary Introduction to Elliptic Curves", 1988, chapter 3
90  """

94  def __init__(self, x_factor, y_factor=None):
95  """
96  Create a new polynomial from the given coefficient lists @p x_factor
97  and @p y_factor.
98 
99  Polynomials @f$ s @f$ over elliptic curves have a canonical form
100  @f$ s(x, y) = a(x) + y\cdot b(x) @f$, where @f$ a, b @f$ are
101  polynomials of @f$ x @f$ alone. The parameter @p x_factor
102  determines the coefficients of @f$ a @f$, @p y_factor determines the
103  coefficients of @f$ b @f$.
104 
105  @param x_factor An iterable that lists the coefficients of the
106  polynomial @f$ a @f$ in ascending order:
107  @p x_factor[0] is the constant,
108  @p x_factor[1] the linear, @p x_factor[2] the
109  quadratic coeffiecient; and so on.
110  @param y_factor Similar to @p x_factor, but for the polynomial
111  @f$ b @f$.
112 
113  For example, construct the polynomial @f$ x^3 + 4x - y\cdot(3x + 2) @f$
114  as follows:
115  @code
116  # Suppose E is an is an EllipticCurve template specialization
117  R = CurvePolynomials( E )
118  p = R( (0, 4, 0, 1), (2, 3) )
119  @endcode
120 
121  @note Leading zeros in the coefficient lists will be ignored.
122  """
123  # x_factor: polynomial in x only
124  # y_factor: polynomial in x [implicitly * y]
125  if isinstance(x_factor, self.__class__):
126  # Casting an object that already has this type means copying it.
127  self.__x_factor = x_factor.__x_factor
128  self.__y_factor = x_factor.__y_factor
129  else:
130  self.__x_factor = self.polynomial_ring()( x_factor )
131  if y_factor is None:
132  self.__y_factor = self.polynomial_ring().zero()
133  else:
134  self.__y_factor = self.polynomial_ring()( y_factor )

157  def leading_coefficient(self):
158  """
159  Return the leading coefficient, or zero if @p self is
160  the zero polynomial.
161 
162  @note The fundamental relation @f$ y^2 = x^3 + Ax + B @f$ forces the
163  linear polynomials @f$ x @f$ and @f$ y @f$ to have different
164  degrees. Thus @f$ \deg(x) = 2 @f$ and @f$ \deg(y) = 3 @f$.
165  The leading coefficient is the coefficient of the power with
166  the highest degree; for @f$ x^4 + 2y^3 @f$ it is 2. (Ignore
167  the reduction of @f$ y^2 @f$ for a moment.)
168  """
169  # Equality is impossible: we compare an odd and an even number.
170  if (2 * self.__x_factor.degree()) > (3 + 2*self.__y_factor.degree()):
171  return self.__x_factor.leading_coefficient()
172  else:
173  return self.__y_factor.leading_coefficient()

205  def __add__(self, other):
206  """
207  Return the sum of @p self and @p other. The infix operator @c + calls
208  this method.
209 
210  Polynomials add coefficient-wise without carrying: the sum of two
211  polynomials @f$ \sum_{k,l} a_{k,l}x^{k}y^{l} @f$ and
212  @f$ \sum_{k,l} b_{k,l}x^{k}y^{l} @f$ is the polynomial
213  @f$ \sum_{k,l} (a_{k,l} + b_{k,l})x^{k}y^{l} @f$.
214  """
215  x = self.__x_factor + other.x_factor()
216  y = self.__y_factor + other.y_factor()

234  def __mul__(self, other):
235  """
236  Return the product of @p self and @p other. The infix operator @c *
237  calls this method.
238 
239  Two polynomials are multiplied through convolution of their
240  coefficients; however that is overly complicated to write with sums.
241  Instead, simply imagine mutliplying the canonical forms
242  @f$ a(x) + y \cdot b(x) @f$ and @f$ c(x) + y \cdot d(x) @f$.
243 
244  @see rings.polynomials.naive.Polynomials.__mul__() for an explanation
245  of univariate polynomial multiplication.
246  """
247  x = self.__x_factor * other.x_factor()
248  xy = self.__x_factor * other.y_factor()
249  yx = self.__y_factor * other.x_factor()
250  y = xy + yx
251 
252  # Reduce y**2 to y2_reduction
253  if self.__y_factor and other.y_factor():
254  #y = self.__y_factor - self.__y_factor # Zero of unknown type
255  x += self.__y_factor * other.y_factor() * self.y2_reduction()
256 
257  return self.__class__( x, y )

259  def __divmod__(self, other):
260  """
261  Return the quotient and remainder of @p self divided by @p other.
262 
263  The built-in method @c divmod() calls this method. For the returned
264  values @c quotient and @c remainder, we have
265  @code
266  self == quotient * other + remainder
267  @endcode
268 
269  @note Division has limited functionality in this implementation. It
270  only works in the following cases:
271  - The divisor (@p other) is a polynomial in @f$ x @f$ alone; this
272  works for all dividends.
273  - Dividend (@p self) and divisor (@p other) are polynomials in
274  @f$ y @f$ alone.
275  """
276  if not other:
277  raise ZeroDivisionError
278 
279  if not self:
280  return self.polynomial_ring().zero(), self.polynomial_ring().zero()
281 
282  if other.y_factor() and (self.__x_factor or other.x_factor()):
283  # Ok, this is cheating. Multivariate polynomial division is,
284  # however, quite complicated and unnecessary for our purpose.
285  raise ValueError("multivariate division is unsupported")
286 
287  if other.x_factor():
288  qx, rx = divmod( self.__x_factor, other.x_factor() )
289  qy, ry = divmod( self.__y_factor, other.x_factor() )
290  else:
291  # Case: self and other have only y factors
292  qx, rx = divmod( self.__y_factor, other.y_factor() )
293  zero = self.polynomial_ring().zero().x_factor()
294  qy, ry = zero, zero
295 
296  quotient = self.__class__( qx, qy )
297  remainder = self.__class__( rx, ry )
298 
299  return quotient, remainder
