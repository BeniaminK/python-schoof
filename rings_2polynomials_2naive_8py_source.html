35  """
36  A ring of polynomials in one indeterminate with coefficients from the
37  provided field; the polynomials support infix notation for ring operations,
38  as well as mixed argument types.
39 
40  This is a template class that must be instantiated with the coefficient
41  field. Use it, for example, as follows:
42  @code
43  # Instantiate the template; GF7x is a class (here: polynomials over GF7)
44  GF7 = fields.finite.naive.FiniteField( 7 )
45  GF7x = Polynomials( GF7 )
46  p = GF7x(1, 1) # Create a polynomial: x + 1
47  q = GF7x(-1, 1) # Another polynomial: x - 1
48  p * q == GF7x(-1, 0, 1) # This is true: (x + 1)(x - 1) = x^2 - 1
49  type(p) is GF7x # This is also true
50 
51  # Use the polynomial 'x' for a more natural syntax
52  x = GF7(0, 1)
53  x**3 - 2*x + 1 == GF7x(1, -2, 0, 1)
54  @endcode
55 
56  The polynomial ring operations are defined in terms of the coefficient
57  field operations. For everything to work as expected, the template
58  argument must be a class whose instances implement the fields.Field
59  interface. Typically, this will be some specialization of
60  fields.finite.naive.FiniteField.
61 
62  Rings, however, also work as a source of coefficients. Only division
63  requires that the leading coefficient is a unit.
64 
65  @note The implementation emphasizes simplicity over speed; it omits
66  possible optimizations.
67 
68  @note The class uses the operand_casting() decorator: @c other operands in
69  binary operations will first be treated as Polynomial elements.
70  If that fails, the operation will be repeated after @p other was fed
71  to the constructor __init__(). If that fails, too, then the
72  operation returns @c NotImplemented (so that @p other.__rop__()
73  might be invoked).
74 
75  @see For example, Robinson, Derek J. S.,
76  "An Introduction to Abstract Algebra", p. 100.

81  def __init__(self, element_description, *further_coefficients):
82  """
83  Create a new polynomial from the given @p element_description.
84 
85  @param element_description
86  Valid input values are:
87  - A Polynomial over the same field: then the polynomial is copied
88  and @p further_coefficients is ignored.
89  - An iterable object (with an @c __iter__() method): then the
90  polynomial coefficients are read from the iterable and, again,
91  @p further_coefficients is ignored. The list must be in
92  ascending order: first the constant, then the linear, quadratic,
93  and cubic coefficients; and so on.
94  - Any number of arguments that can be interpreted as coefficient
95  field elements; they will be combined
96  into the list of coefficients
97 
98  @param further_coefficients The list of variable positional
99  arguments.
100 
101  For example, there are several ways to construct the polynomial
102  @f$ x^3 + 4x^2 - 3x + 2@f$:
103  @code
104  # Suppose R is a Polynomials template specialization
105  # List of coefficients
106  p2 = R( [2, -3, 4, 1] )
107  # Variable argument count
108  p3 = R( 2, -3, 4, 1 )
109  # Copy
110  p1 = R( p2 )
111  @endcode
112 
113  @note Leading zeros will be ignored.
114  """
115  # List of coefficients is in ascending order without leading zeros.
116  # Example: x^2 + 2x + 5 = [5, 2, 1]
117  if isinstance(element_description, self.__class__):
118  # A reference suffices because objects are immutable
119  self.__coefficients = element_description.__coefficients

121  else:

123  coefficients = element_description
124  else:

206  def __add__(self, other):
207  """
208  Return the sum of @p self and @p other. The infix operator @c + calls
209  this method.
210 
211  Polynomials add coefficient-wise without carrying: the sum of two
212  polynomials @f$ \sum_{k} a_{k}x^{k} @f$ and
213  @f$ \sum_{k} b_{k}x^{k} @f$ is the polynomial
214  @f$ \sum_{k} (a_{k} + b_{k})x^{k} @f$.
215  """
216  zero = self._coefficient_field.zero()
217  coefficient_pairs = self.__pad_and_zip(
218  self.__coefficients,
219  other.__coefficients,
220  zero
221  )
222  coefficient_sums = [ x + y for x, y in coefficient_pairs ]
223  return self.__class__( coefficient_sums )

238  def __mul__(self, other):
239  """
240  Return the product of @p self and @p other. The infix operator @c *
241  calls this method.
242 
243  Two polynomials are multiplied through convolution of their
244  coefficients: for polynomials @f$ \sum_{k} a_{k}x^{k} @f$ and
245  @f$ \sum_{k} b_{k}x^{k} @f$, their product is the polynomial
246  @f$ \sum_{k} \sum_{j=0}^{k}(a_{j} + b_{k-j})x^{k} @f$.
247  """
248  # Initialize result as list of all zeros
249  zero = self._coefficient_field.zero()
250  # Add 2 because degrees count from 0.
251  result = [ zero ] * (self.degree() + other.degree() + 2)
252 
253  for i, x in enumerate(self.__coefficients):
254  for j, y in enumerate(other.__coefficients):
255  result[i + j] += x * y
256 
257  return self.__class__( result )

260  def __divmod__(self, other):
261  """
262  Return the quotient and remainder of @p self divided by @p other.
263 
264  The built-in method @c divmod() calls this method. For the returned
265  values @c quotient and @c remainder, we have
266  @code
267  self == quotient * other + remainder
268  @endcode
269  """
270  # Lists will be modified, so copy them
271  dividend = self.__coefficients[:]
272  divisor = other.__coefficients[:]
273  n = other.degree()
274 
275  zero = self._coefficient_field.zero()
276  quotient = [ zero ] * (self.degree() - n + 1)
277 
278  for k in reversed(range( 0, len(quotient) )):
279  quotient[k] = dividend[n + k] / divisor[n]
280  for j in range(k, n + k):
281  dividend[j] -= quotient[k] * divisor[j - k]
282 
283  remainder = dividend[ 0 : n ]
284 
285  return self.__class__( quotient ), \
286  self.__class__( remainder )

289  def __call__(self, point):
290  """
291  Return the polynomial function's value at @p point.
292 
293  This method evaluates the polynomial by replacing the indeterminate
294  with @p point and summing the powers: for a polynomial
295  @f$ p(x) = \sum_{k} a_{k}x^{k} @f$, the value at @f$ c @f$ is
296  @f$ p(c) = \sum_{k} a_{k}c^{k} @f$.
297  """
298  return sum( [ c * point**i for i, c in enumerate(self.__coefficients) ] )

341  @staticmethod
342  def __pad_and_zip(list1, list2, padding_element):
343  """
344  Combine @p list1 and @p list1 into a single list of pairs. If the
345  lists have different lengths, pad the shorter list with
346  @p padding_element: fill in elements until they have the same length.
347  """
348  max_length = max( len(list1), len(list2) )
349  padded_list1 = list1 + ( [padding_element] * (max_length - len(list1)) )
350  padded_list2 = list2 + ( [padding_element] * (max_length - len(list2)) )
351  return zip( padded_list1, padded_list2 )
